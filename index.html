function checkReorderAnswer() { if (appState.isPopupShowing) return; const q = appState.currentQuestion; const userStrRaw = appState.currentReorderAnswer.join('').toLowerCase().replace(/\s+/g, ''); const userChoiceDisplay = appState.currentReorderAnswer.join(' '); appState.isPopupShowing = true; if (userStrRaw === q.correctStr) { handleCorrect(userChoiceDisplay); } else { document.getElementById('reorder-answer-zone').classList.add('shake-animation', 'border-red-400', 'bg-red-50'); setTimeout(() => document.getElementById('reorder-answer-zone').classList.remove('shake-animation', 'border-red-400', 'bg-red-50'), 500); handleWrong(userChoiceDisplay); } } function handleCorrect(userChoiceStr) { const q = appState.currentQuestion; if (!appState.answers[q.id]) appState.answers[q.id] = { firstTry: true }; appState.answers[q.id].isDone = true; // Đánh dấu là đã hoàn thành vĩnh viễn showFeedback(true, getExplanation(q, true, userChoiceStr)); fireConfetti(); } function handleWrong(userChoiceStr) { const q = appState.currentQuestion; // Action khi bấm nút trên Popup function closeFeedbackAction(isCorrect) { document.getElementById('feedback-card').classList.add('scale-90'); setTimeout(() => { document.getElementById('feedback-overlay').classList.add('hidden'); appState.isPopupShowing = false; if (isCorrect) { appState.pendingQueue.shift(); } else { const q = appState.pendingQueue.shift(); appState.pendingQueue.push(q); } updateUI(); renderNextQuestion(); }, 150); } function updateUI() { const testId = appState.testIds[appState.currentTestIndex]; const testQuestions = appState.questionsByTest[testId]; const total = testQuestions.length; // Tính số câu đã làm xong dựa trên trạng thái isDone let completed = 0; testQuestions.forEach(q => { if (appState.answers[q.id] && appState.answers[q.id].isDone) completed++; }); document.getElementById('header-test-info').innerText = `Bài ${testId} - Tiến độ:`; document.getElementById('header-progress-text').innerText = `${completed}/${total}`; document.getElementById('progress-bar').style.width = `${(completed/total)*100}%`; document.getElementById('quick-nav-title').innerText = `Danh sách câu hỏi - Bài ${testId}`; renderQuickNavGrid(testId, testQuestions); } function toggleTestNav() { document.getElementById('test-nav-modal').classList.toggle('hidden'); document.getElementById('quick-nav-modal').classList.add('hidden'); // Đóng menu kia renderTestNavList(); } function renderTestNavList() { const list = document.getElementById('test-nav-list'); list.innerHTML = ''; appState.testIds.forEach((testId, index) => { const testQuestions = appState.questionsByTest[testId]; let doneCount = 0; testQuestions.forEach(q => { if (appState.answers[q.id] && appState.answers[q.id].isDone) doneCount++; }); const btn = document.createElement('button'); const isCurrent = index === appState.currentTestIndex; btn.className = `p-3 rounded-xl font-bold text-left flex justify-between items-center transition-colors btn-press ${isCurrent ? 'bg-indigo-500 text-white shadow-md' : 'bg-gray-50 text-gray-700 hover:bg-gray-100 border-2 border-gray-100'}`; btn.innerHTML = `Bài ${testId} ${doneCount}/30`; btn.onclick = () => { toggleTestNav(); if (!isCurrent) startTest(index); }; list.appendChild(btn); }); } function toggleQuickNav() { document.getElementById('quick-nav-modal').classList.toggle('hidden'); document.getElementById('test-nav-modal').classList.add('hidden'); // Đóng menu kia } function renderQuickNavGrid(testId, testQuestions) { const grid = document.getElementById('quick-nav-grid'); grid.innerHTML = ''; testQuestions.forEach((q, i) => { const isDone = appState.answers[q.id] && appState.answers[q.id].isDone; const btn = document.createElement('button'); if (isDone) { btn.className = `p-2 rounded-lg font-bold text-sm bg-[#22c55e] text-white shadow-md cursor-default opacity-80`; btn.innerText = i + 1; } else { btn.className = `p-2 rounded-lg font-bold text-sm btn-press bg-gray-100 text-gray-500 hover:bg-gray-200`; btn.innerText = i + 1; btn.onclick = () => { // Tìm và đẩy câu hỏi này lên đầu hàng đợi const qIndexInQueue = appState.pendingQueue.findIndex(item => item.id === q.id); if (qIndexInQueue > -1) { const selectedQ = appState.pendingQueue.splice(qIndexInQueue, 1)[0]; appState.pendingQueue.unshift(selectedQ); renderNextQuestion(); toggleQuickNav(); } }; } grid.appendChild(btn); }); } function startTest(index) { appState.currentTestIndex = index; const testId = appState.testIds[index]; // Khởi tạo hàng đợi: CHỈ lấy những câu chưa được làm đúng appState.pendingQueue = appState.questionsByTest[testId].filter(q => { return !(appState.answers[q.id] && appState.answers[q.id].isDone); }); renderNextQuestion(); updateUI(); } // TỔNG KẾT THEO TỪNG BÀI (TEST)
